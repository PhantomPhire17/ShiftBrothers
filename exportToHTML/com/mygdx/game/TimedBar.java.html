package com.mygdx.game;

import com.badlogic.gdx.graphics.glutils.ShapeRenderer;

public class TimedBar extends Rect implements Constants {

    private int frames;
    private int x;
    private int y;
    private int width;
    private int height;
    private Direction dir;
    private int frameCounter = 0;


    public TimedBar() {
        dir = Direction.FORWARDS;
    }

    public void setTimer(int frames) {
        this.frames = frames;
    }

    public void drawTimedBar(ShapeRenderer shape) {
        while (frameCounter <= frames) {
            frameCounter++;
            setDimensions(getRectX(), getRectY(), getRectWidth() - getRemovedWidthPerFrame() * frameCounter, getRectHeight());
            draw(shape);
        }
    }

    public int getRemovedWidthPerFrame() {
        return getRectWidth()/frames;
    }

    public void setDirection(Direction dir) {
        this.dir = dir;
    }

    public enum Direction {
        FORWARDS, BACKWARDS
    }
}
