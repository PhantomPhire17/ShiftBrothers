package com.mygdx.game;

import com.badlogic.gdx.graphics.Color;
import com.badlogic.gdx.graphics.glutils.ShapeRenderer;

public class Ball {

    private int x;
    private int y;
    private int ex;
    private int ey;
    private int radius;
    private Color color;
    private Ability ab;

    public Ball() {}
    public Ball(int x, int y, int radius) {
        this.x = x;
        this.y = y;
        this.radius = radius;
    }

    //---------------------------------------
    //---------------------------------------
    //---------------------------------------
    //---------------------------------------

    // 1. GETTERS

    //---------------------------------------
    //---------------------------------------
    //---------------------------------------
    //---------------------------------------

    //---------------------------------------
    //---------------------------------------
    //---------------------------------------
    //---------------------------------------

    public Ability getAbility() {
        return ab;
    }
    public boolean isCircle() {
        boolean XisRightOfBorder = ex > x;
        boolean XisLeftOfBorder = ex < x + radius*2;
        boolean YisHigherThanBorder = ey > y - radius*2 + radius;
        boolean YisLowerThanBorder = ey < y;
        if (XisRightOfBorder && XisLeftOfBorder && YisHigherThanBorder && YisLowerThanBorder) {
            return true;
        }
        return false;
    }

    // 2. SETTERS

    //---------------------------------------
    //---------------------------------------
    //---------------------------------------
    //---------------------------------------
    public void setColor(Color color) {
        this.color = color;
    }
    public void setDimensions(int x, int y, int width, int height) {
        this.x = x;
        this.y = y;
        this.radius = radius;
    }
    public void setMouseListeners(int ex, int ey) {
        this.ex = ex;
        this.ey = ey;
    }

    public void setAbility(Ability ab) {
        this.ab = ab;
    }

    //---------------------------------------
    //---------------------------------------
    //---------------------------------------
    //---------------------------------------

    // 3. DRAW

    //---------------------------------------
    //---------------------------------------
    //---------------------------------------
    //---------------------------------------

    public void draw(ShapeRenderer shape) {
        shape.setColor(color);
        shape.circle(x,y,radius);
    }

    //---------------------------------------
    //---------------------------------------
    //---------------------------------------
    //---------------------------------------

    // 4. GENERAL

    //---------------------------------------
    //---------------------------------------
    //---------------------------------------
    //---------------------------------------
}
